    <script>
        // WebSocket Client Class
        class WebSocketClient {
            constructor(role = 'teacher', language = 'en-US', ttsServiceType = 'browser') {
                this.ws = null;
                this.sessionId = null;
                this.role = role;
                this.language = language;
                this.ttsServiceType = ttsServiceType;
                this.connected = false;
                this.eventListeners = {
                    'open': [],
                    'close': [],
                    'error': [],
                    'message': [],
                    'status': [],
                    'translation': []
                };
            }
            
            connect() {
                if (this.ws && (this.ws.readyState === WebSocket.CONNECTING || this.ws.readyState === WebSocket.OPEN)) {
                    this.notifyListeners('status', 'Already connected or connecting');
                    return;
                }
                
                const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
                const host = window.location.host;
                const wsUrl = `${protocol}//${host}/ws`;
                
                this.notifyListeners('status', `Connecting to ${wsUrl}...`);
                
                try {
                    this.ws = new WebSocket(wsUrl);
                    
                    this.ws.onopen = () => {
                        this.connected = true;
                        this.notifyListeners('status', 'Connected');
                        this.notifyListeners('open');
                        this.register(this.role, this.language);
                    };
                    
                    this.ws.onmessage = (event) => {
                        try {
                            const message = JSON.parse(event.data);
                            this.notifyListeners('message', message);
                            
                            if (message.type === 'connection' && message.sessionId) {
                                this.sessionId = message.sessionId;
                                this.notifyListeners('status', `Session ID: ${this.sessionId}`);
                            }
                            
                            if (message.type === 'translation') {
